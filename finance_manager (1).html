Cấu hình Google Sheets

Hướng dẫn:
1. Mở Google Sheets → Tạo bảng tính mới
2. Sao chép Spreadsheet ID từ URL (phần giữa /d/ và /edit)
3. Vào Extensions → Apps Script → Dán đoạn script bên dưới và Deploy as Web App
4. Nhập Web App URL vào ô dưới

Google Apps Script (Copy & Deploy)

function doGet(e) { var p = e.parameter; var type = p.type; var ss = SpreadsheetApp.getActiveSpreadsheet(); try { if (type === "transaction") { var sh = ss.getSheetByName("GiaoDich") || ss.insertSheet("GiaoDich"); if (sh.getLastRow() === 0) sh.appendRow(["Ngay","Mo ta","Danh muc","Loai","So tien"]); sh.appendRow([p.date, p.desc, p.category, p.txtype, Number(p.amount)]); } else if (type === "budget") { var sh = ss.getSheetByName("NganSach") || ss.insertSheet("NganSach"); if (sh.getLastRow() === 0) sh.appendRow(["Danh muc","Gioi han","Da chi"]); var data = sh.getDataRange().getValues(); var found = false; for (var i = 1; i < data.length; i++) { if (data[i][0] === p.category) { sh.getRange(i+1,2,1,2).setValues([[Number(p.limit),Number(p.spent)]]); found = true; break; } } if (!found) sh.appendRow([p.category, Number(p.limit), Number(p.spent)]); } else if (type === "goal") { var sh = ss.getSheetByName("MucTieu") || ss.insertSheet("MucTieu"); if (sh.getLastRow() === 0) sh.appendRow(["Ten muc tieu","So tien muc tieu","Da tiet kiem"]); sh.appendRow([p.name, Number(p.target), Number(p.current)]); } else if (type === "debt") { var sh = ss.getSheetByName("No") || ss.insertSheet("No"); if (sh.getLastRow() === 0) sh.appendRow(["Ten khoan no","Tong no","Da tra","Nguoi cho vay","Con lai"]); sh.appendRow([p.name, Number(p.total), Number(p.paid), p.lender, Number(p.total)-Number(p.paid)]); } return ContentService .createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:"error",msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
