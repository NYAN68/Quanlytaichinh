Google スプレッドシート設定

設定手順：
1. Google スプレッドシート を開いて新しいシートを作成
2. URLから Spreadsheet ID をコピー（/d/ と /edit の間の文字列）
3. 拡張機能 → Apps Script を開いて下のスクリプトを貼り付け → ウェブアプリとしてデプロイ
4. 発行されたWeb App URLを下の入力欄に貼り付け

Google Apps Script（コピー＆デプロイ）

function doGet(e) { var p = e.parameter; var type = p.type; var month = p.month || ""; // e.g. "2025_04" var ss = SpreadsheetApp.getActiveSpreadsheet(); try { if (type === "transaction") { var shName = month ? "取引_" + month : "取引"; var sh = ss.getSheetByName(shName) || ss.insertSheet(shName); if (sh.getLastRow() === 0) sh.appendRow(["日付","説明","カテゴリ","種類","金額"]); sh.appendRow([p.date, p.desc, p.category, p.txtype, Number(p.amount)]); } else if (type === "budget") { var shName = month ? "予算_" + month : "予算"; var sh = ss.getSheetByName(shName) || ss.insertSheet(shName); if (sh.getLastRow() === 0) sh.appendRow(["カテゴリ","上限","使用済み"]); var data = sh.getDataRange().getValues(); var found = false; for (var i = 1; i < data.length; i++) { if (data[i][0] === p.category) { sh.getRange(i+1,2,1,2).setValues([[Number(p.limit),Number(p.spent)]]); found = true; break; } } if (!found) sh.appendRow([p.category, Number(p.limit), Number(p.spent)]); } else if (type === "goal") { var shName = month ? "目標_" + month : "目標"; var sh = ss.getSheetByName(shName) || ss.insertSheet(shName); if (sh.getLastRow() === 0) sh.appendRow(["目標名","目標金額","貯金済み"]); sh.appendRow([p.name, Number(p.target), Number(p.current)]); } else if (type === "debt") { var shName = month ? "借金_" + month : "借金"; var sh = ss.getSheetByName(shName) || ss.insertSheet(shName); if (sh.getLastRow() === 0) sh.appendRow(["借金名","合計","返済済み","貸主","残高"]); sh.appendRow([p.name, Number(p.total), Number(p.paid), p.lender, Number(p.total)-Number(p.paid)]); } return ContentService .createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:"error",msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
